// src/lib/bus.ts
type Handler = (p: any) => void;
const map = new Map<string, Set<Handler>>();
export const bus = {
  on(ev: string, h: Handler) {
    (map.get(ev) ?? map.set(ev, new Set()).get(ev))!.add(h);
    return () => map.get(ev)!.delete(h);
  },
  emit(ev: string, p?: any) {
    map.get(ev)?.forEach(h => h(p));
  }
};
